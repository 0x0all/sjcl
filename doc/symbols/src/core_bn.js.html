 18   /**
 19    * Initializes this with it, either as a bn, a number, or a hex string.
 20    */
 21   initWith: function(it) {
 22     var i=0, k, n, l;
 23     switch(typeof it) {
 24     case "object":
 25       this.limbs = it.limbs.slice(0);
 26       break;
 27       
 28     case "number":
 29       this.limbs = [it];
 30       this.normalize();
 31       break;
 32       
 33     case "string":
 34       it = it.replace(/^0x/, '');
 35       this.limbs = [];
 36       // hack
 37       k = this.radix / 4;
 38       for (i=0; i < it.length; i+=k) {
 39         this.limbs.push(parseInt(it.substring(Math.max(it.length - i - k, 0), it.length - i),16));
 40       }
 41       break;
 42 
 43     default:
 44       this.limbs = [0];
 45     }
 46     return this;

 49   /**
 50    * Returns true if "this" and "that" are equal.  Calls fullReduce().
 51    * Equality test is in constant time.
 52    */
 53   equals: function(that) {
 54     if (typeof that === "number") { that = new this._class(that); }
 55     var difference = 0, i;
 56     this.fullReduce();
 57     that.fullReduce();
 58     for (i = 0; i < this.limbs.length || i < that.limbs.length; i++) {
 59       difference |= this.getLimb(i) ^ that.getLimb(i);
 60     }
 61     return (difference === 0);
 62   },

 71   /**
 72    * Constant time comparison function.
 73    * Returns 1 if this >= that, or zero otherwise.
 74    */
 75   greaterEquals: function(that) {
 76     if (typeof that === "number") { that = new this._class(that); }
 77     var less = 0, greater = 0, i, a, b;
 78     i = Math.max(this.limbs.length, that.limbs.length) - 1;
 79     for (; i>= 0; i--) {
 80       a = this.getLimb(i);
 81       b = that.getLimb(i);
 82       greater |= (b - a) & ~less;
 83       less |= (a - b) & ~greater;
 84     }
 85     return (greater | ~less) >>> 31;
 86   },

104   /** this += that.  Does not normalize. */
105   addM: function(that) {
106     if (typeof(that) !== "object") { that = new this._class(that); }
107     var i, l=this.limbs, ll=that.limbs;
108     for (i=l.length; i<ll.length; i++) {

117   /** this *= 2.  Requires normalized; ends up normalized. */
118   doubleM: function() {
119     var i, carry=0, tmp, r=this.radix, m=this.radixMask, l=this.limbs;
120     for (i=0; i<l.length; i++) {
121       tmp = l[i];
122       tmp = tmp+tmp+carry;
123       l[i] = tmp & m;
124       carry = tmp >> r;
125     }
126     if (carry) {
127       l.push(carry);
128     }
129     return this;
130   },

132   /** this /= 2, rounded down.  Requires normalized; ends up normalized. */
133   halveM: function() {
134     var i, carry=0, tmp, r=this.radix, l=this.limbs;
135     for (i=l.length-1; i>=0; i--) {
136       tmp = l[i];
137       l[i] = (tmp+carry)>>1;
138       carry = (tmp&1) << r;
139     }
140     if (!l[l.length-1]) {
141       l.pop();
142     }
143     return this;
144   },

146   /** this -= that.  Does not normalize. */
147   subM: function(that) {
148     if (typeof(that) !== "object") { that = new this._class(that); }
149     var i, l=this.limbs, ll=that.limbs;
150     for (i=l.length; i<ll.length; i++) {
151       l[i] = 0;
152     }
153     for (i=0; i<ll.length; i++) {
154       l[i] -= ll[i];
155     }
156     return this;

159   mod: function(that) {
160     var neg = !this.greaterEquals(new sjcl.bn(0));
161     
162     that = new sjcl.bn(that).normalize(); // copy before we begin
163     var out = new sjcl.bn(this).normalize(), ci=0;
164     
165     if (neg) out = (new sjcl.bn(0)).subM(out).normalize();
166     
167     for (; out.greaterEquals(that); ci++) {
168       that.doubleM();
169     }
170     
171     if (neg) out = that.sub(out).normalize();
172     
173     for (; ci > 0; ci--) {
174       that.halveM();
175       if (out.greaterEquals(that)) {
176         out.subM(that).normalize();
177       }
178     }
179     return out.trim();
180   },

182   /** return inverse mod prime p.  p must be odd. Binary extended Euclidean algorithm mod p. */
183   inverseMod: function(p) {
184     var a = new sjcl.bn(1), b = new sjcl.bn(0), x = new sjcl.bn(this), y = new sjcl.bn(p), tmp, i, nz=1;
185     
186     if (!(p.limbs[0] & 1)) {
187       throw (new sjcl.exception.invalid("inverseMod: p must be odd"));
188     }
189     
190     // invariant: y is odd
191     do {
192       if (x.limbs[0] & 1) {
193         if (!x.greaterEquals(y)) {
194           // x < y; swap everything
195           tmp = x; x = y; y = tmp;
196           tmp = a; a = b; b = tmp;
197         }
198         x.subM(y);
199         x.normalize();
200         
201         if (!a.greaterEquals(b)) {
202           a.addM(p);
203         }
204         a.subM(b);
205       }
206       
207       // cut everything in half
208       x.halveM();
209       if (a.limbs[0] & 1) {
210         a.addM(p);
211       }
212       a.normalize();
213       a.halveM();
214       
215       // check for termination: x ?= 0
216       for (i=nz=0; i<x.limbs.length; i++) {
217         nz |= x.limbs[i];
218       }
219     } while(nz);
220     
221     if (!y.equals(1)) {
222       throw (new sjcl.exception.invalid("inverseMod: p and x must be relatively prime"));
223     }
224     
225     return b;
226   },

238   /** this * that.  Normalizes and reduces. */
239   mul: function(that) {
240     if (typeof(that) === "number") { that = new this._class(that); }
241     var i, j, a = this.limbs, b = that.limbs, al = a.length, bl = b.length, out = new this._class(), c = out.limbs, ai, ii=this.maxMul;
242 
243     for (i=0; i < this.limbs.length + that.limbs.length + 1; i++) {
244       c[i] = 0;
245     }
246     for (i=0; i<al; i++) {
247       ai = a[i];
248       for (j=0; j<bl; j++) {
249         c[i+j] += ai * b[j];
250       }
251      
252       if (!--ii) {
253         ii = this.maxMul;
254         out.cnormalize();
255       }
256     }
257     return out.cnormalize().reduce();
258   },

265   /** this ^ n.  Uses square-and-multiply.  Normalizes and reduces. */
266   power: function(l) {
267     if (typeof(l) === "number") {
268       l = [l];
269     } else if (l.limbs !== undefined) {
270       l = l.normalize().limbs;
271     }
272     var i, j, out = new this._class(1), pow = this;
273 
274     for (i=0; i<l.length; i++) {
275       for (j=0; j<this.radix; j++) {
276         if (l[i] & (1<<j)) {
277           out = out.mul(pow);
278         }
279         pow = pow.square();
280       }
281     }
282     
283     return out;
284   },

291   /** this ^ x mod N */
292   powermod: function(x, N) {
293     var result = new sjcl.bn(1), a = new sjcl.bn(this), k = new sjcl.bn(x);
294     while (true) {
295       if (k.limbs[0] & 1) { result = result.mulmod(a, N); }
296       k.halveM();
297       if (k.equals(0)) { break; }
298       a = a.mulmod(a, N);
299     }
300     return result.normalize().reduce();

322   /** Propagate carries. */
323   normalize: function() {
324     var carry=0, i, pv = this.placeVal, ipv = this.ipv, l, m, limbs = this.limbs, ll = limbs.length, mask = this.radixMask;
325     for (i=0; i < ll || (carry !== 0 && carry !== -1); i++) {
326       l = (limbs[i]||0) + carry;
327       m = limbs[i] = l & mask;
328       carry = (l-m)*ipv;
329     }
330     if (carry === -1) {
331       limbs[i-1] -= this.placeVal;
332     }
333     return this;
334   },

336   /** Constant-time normalize. Does not allocate additional space. */
337   cnormalize: function() {
338     var carry=0, i, ipv = this.ipv, l, m, limbs = this.limbs, ll = limbs.length, mask = this.radixMask;
339     for (i=0; i < ll-1; i++) {
340       l = limbs[i] + carry;
341       m = limbs[i] = l & mask;
342       carry = (l-m)*ipv;
343     }
344     limbs[i] += carry;
345     return this;
346   },

348   /** Serialize to a bit array */
349   toBits: function(len) {
350     this.fullReduce();
351     len = len || this.exponent || this.bitLength();
352     var i = Math.floor((len-1)/24), w=sjcl.bitArray, e = (len + 7 & -8) % this.radix || this.radix,
353         out = [w.partial(e, this.getLimb(i))];
354     for (i--; i >= 0; i--) {
355       out = w.concat(out, [w.partial(Math.min(this.radix,len), this.getLimb(i))]);
356       len -= this.radix;
357     }
358     return out;
359   },

361   /** Return the length in bits, rounded up to the nearest byte. */
362   bitLength: function() {
363     this.fullReduce();
364     var out = this.radix * (this.limbs.length - 1),
365         b = this.limbs[this.limbs.length - 1];
366     for (; b; b >>>= 1) {
367       out ++;
368     }
369     return out+7 & -8;

394 /**
395  * Creates a new subclass of bn, based on reduction modulo a pseudo-Mersenne prime,
396  * i.e. a prime of the form 2^e + sum(a * 2^b),where the sum is negative and sparse.
397  */
398 sjcl.bn.pseudoMersennePrime = function(exponent, coeff) {
399   /** @constructor 
400   * @private
401   */
402   function p(it) {
403     this.initWith(it);
404     /*if (this.limbs[this.modOffset]) {
405       this.reduce();
406     }*/
407   }
408 
409   var ppr = p.prototype = new sjcl.bn(), i, tmp, mo;
410   mo = ppr.modOffset = Math.ceil(tmp = exponent / ppr.radix);
411   ppr.exponent = exponent;
412   ppr.offset = [];
413   ppr.factor = [];
414   ppr.minOffset = mo;
415   ppr.fullMask = 0;
416   ppr.fullOffset = [];
417   ppr.fullFactor = [];
418   ppr.modulus = p.modulus = new sjcl.bn(Math.pow(2,exponent));
419   
420   ppr.fullMask = 0|-Math.pow(2, exponent % ppr.radix);
421 
422   for (i=0; i<coeff.length; i++) {
423     ppr.offset[i] = Math.floor(coeff[i][0] / ppr.radix - tmp);
424     ppr.fullOffset[i] = Math.ceil(coeff[i][0] / ppr.radix - tmp);
425     ppr.factor[i] = coeff[i][1] * Math.pow(1/2, exponent - coeff[i][0] + ppr.offset[i] * ppr.radix);
426     ppr.fullFactor[i] = coeff[i][1] * Math.pow(1/2, exponent - coeff[i][0] + ppr.fullOffset[i] * ppr.radix);
427     ppr.modulus.addM(new sjcl.bn(Math.pow(2,coeff[i][0])*coeff[i][1]));
428     ppr.minOffset = Math.min(ppr.minOffset, -ppr.offset[i]); // conservative
429   }
430   ppr._class = p;
431   ppr.modulus.cnormalize();
432 
433   /** Approximate reduction mod p.  May leave a number which is negative or slightly larger than p.
434    * @memberof sjcl.bn
435    * @this { sjcl.bn }
436    */
437   ppr.reduce = function() {
438     var i, k, l, mo = this.modOffset, limbs = this.limbs, aff, off = this.offset, ol = this.offset.length, fac = this.factor, ll;
439 
440     i = this.minOffset;
441     while (limbs.length > mo) {
442       l = limbs.pop();
443       ll = limbs.length;
444       for (k=0; k<ol; k++) {
445         limbs[ll+off[k]] -= fac[k] * l;
446       }
447       
448       i--;
449       if (!i) {
450         limbs.push(0);
451         this.cnormalize();
452         i = this.minOffset;
453       }
454     }
455     this.cnormalize();
456 
457     return this;
458   };

460   /** @memberof sjcl.bn
461   * @this { sjcl.bn }
462   */
463   ppr._strongReduce = (ppr.fullMask === -1) ? ppr.reduce : function() {
464     var limbs = this.limbs, i = limbs.length - 1, k, l;
465     this.reduce();
466     if (i === this.modOffset - 1) {
467       l = limbs[i] & this.fullMask;
468       limbs[i] -= l;
469       for (k=0; k<this.fullOffset.length; k++) {
470         limbs[i+this.fullOffset[k]] -= this.fullFactor[k] * l;
471       }
472       this.normalize();
473     }
474   };

476   /** mostly constant-time, very expensive full reduction.
477    * @memberof sjcl.bn
478    * @this { sjcl.bn }
479    */
480   ppr.fullReduce = function() {
481     var greater, i;
482     // massively above the modulus, may be negative
483     
484     this._strongReduce();
485     // less than twice the modulus, may be negative
486 
487     this.addM(this.modulus);
488     this.addM(this.modulus);
489     this.normalize();
490     // probably 2-3x the modulus
491     
492     this._strongReduce();
493     // less than the power of 2.  still may be more than
494     // the modulus
495 
496     // HACK: pad out to this length
497     for (i=this.limbs.length; i<this.modOffset; i++) {
498       this.limbs[i] = 0;
499     }
500     
501     // constant-time subtract modulus
502     greater = this.greaterEquals(this.modulus);
503     for (i=0; i<this.limbs.length; i++) {
504       this.limbs[i] -= this.modulus.limbs[i] * greater;
505     }
506     this.cnormalize();
507 
508     return this;
509   };

545 sjcl.bn.random = function(modulus, paranoia) {
546   if (typeof modulus !== "object") { modulus = new sjcl.bn(modulus); }
547   var words, i, l = modulus.limbs.length, m = modulus.limbs[l-1]+1, out = new sjcl.bn();
548   while (true) {
549     // get a sequence whose first digits make sense
550     do {
551       words = sjcl.random.randomWords(l, paranoia);
552       if (words[l-1] < 0) { words[l-1] += 0x100000000; }
553     } while (Math.floor(words[l-1] / m) === Math.floor(0x100000000 / m));
554     words[l-1] %= m;
555 
556     // mask off all the limbs
557     for (i=0; i<l-1; i++) {
558       words[i] &= modulus.radixMask;
559     }
560 
561     // check the rest of the digitssj
562     out.limbs = words;
563     if (!out.greaterEquals(modulus)) {
564       return out;
565     }
566   }

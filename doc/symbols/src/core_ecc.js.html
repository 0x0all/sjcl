  6 /**
  7  * Represents a point on a curve in affine coordinates.
  8  * @constructor
  9  * @param {sjcl.ecc.curve} curve The curve that this point lies on.
 10  * @param {bigInt} x The x coordinate.
 11  * @param {bigInt} y The y coordinate.
 12  */
 13 sjcl.ecc.point = function(curve,x,y) {
 14   if (x === undefined) {
 15     this.isIdentity = true;
 16   } else {
 17     this.x = x;
 18     this.y = y;
 19     this.isIdentity = false;
 20   }
 21   this.curve = curve;
 22 };
 23

 25 
 26 sjcl.ecc.point.prototype = {
 27   toJac: function() {
 28     return new sjcl.ecc.pointJac(this.curve, this.x, this.y, new this.curve.field(1));
 29   },
 30 
 31   mult: function(k) {
 32     return this.toJac().mult(k, this).toAffine();
 33   },
 34   
 35   /**
 36    * Multiply this point by k, added to affine2*k2, and return the answer in Jacobian coordinates.
 37    * @param {bigInt} k The coefficient to multiply this by.
 38    * @param {bigInt} k2 The coefficient to multiply affine2 this by.
 39    * @param {sjcl.ecc.point} affine The other point in affine coordinates.
 40    * @return {sjcl.ecc.pointJac} The result of the multiplication and addition, in Jacobian coordinates.
 41    */
 42   mult2: function(k, k2, affine2) {
 43     return this.toJac().mult2(k, this, k2, affine2).toAffine();
 44   },
 45   
 46   multiples: function() {
 47     var m, i, j;
 48     if (this._multiples === undefined) {
 49       j = this.toJac().doubl();
 50       m = this._multiples = [new sjcl.ecc.point(this.curve), this, j.toAffine()];
 51       for (i=3; i<16; i++) {
 52         j = j.add(this);
 53         m.push(j.toAffine());
 54       }
 55     }
 56     return this._multiples;
 57   },
 58 
 59   isValid: function() {
 60     return this.y.square().equals(this.curve.b.add(this.x.mul(this.curve.a.add(this.x.square()))));
 61   },
 62 
 63   toBits: function() {
 64     return sjcl.bitArray.concat(this.x.toBits(), this.y.toBits());
 65   }
 66 };

 68 /**
 69  * Represents a point on a curve in Jacobian coordinates. Coordinates can be specified as bigInts or strings (which
 70  * will be converted to bigInts).
 71  *
 72  * @constructor
 73  * @param {bigInt/string} x The x coordinate.
 74  * @param {bigInt/string} y The y coordinate.
 75  * @param {bigInt/string} z The z coordinate.
 76  * @param {sjcl.ecc.curve} curve The curve that this point lies on.
 77  */
 78 sjcl.ecc.pointJac = function(curve, x, y, z) {
 79   if (x === undefined) {
 80     this.isIdentity = true;
 81   } else {
 82     this.x = x;
 83     this.y = y;
 84     this.z = z;
 85     this.isIdentity = false;
 86   }
 87   this.curve = curve;
 88 };
 89 
 90 sjcl.ecc.pointJac.prototype = {
 91   /**
 92    * Adds S and T and returns the result in Jacobian coordinates. Note that S must be in Jacobian coordinates and T must be in affine coordinates.
 93    * @param {sjcl.ecc.pointJac} S One of the points to add, in Jacobian coordinates.
 94    * @param {sjcl.ecc.point} T The other point to add, in affine coordinates.
 95    * @return {sjcl.ecc.pointJac} The sum of the two points, in Jacobian coordinates. 
 96    */
 97   add: function(T) {
 98     var S = this, sz2, c, d, c2, x1, x2, x, y1, y2, y, z;
 99     if (S.curve !== T.curve) {
100       throw("sjcl.ecc.add(): Points must be on the same curve to add them!");
101     }
102 
103     if (S.isIdentity) {
104       return T.toJac();
105     } else if (T.isIdentity) {
106       return S;
107     }
108 
109     sz2 = S.z.square();
110     c = T.x.mul(sz2).subM(S.x);
111 
112     if (c.equals(0)) {
113       if (S.y.equals(T.y.mul(sz2.mul(S.z)))) {
114         // same point
115         return S.doubl();
116       } else {
117         // inverses
118         return new sjcl.ecc.pointJac(S.curve);
119       }
120     }
121     
122     d = T.y.mul(sz2.mul(S.z)).subM(S.y);
123     c2 = c.square();
124 
125     x1 = d.square();
126     x2 = c.square().mul(c).addM( S.x.add(S.x).mul(c2) );
127     x  = x1.subM(x2);
128 
129     y1 = S.x.mul(c2).subM(x).mul(d);
130     y2 = S.y.mul(c.square().mul(c));
131     y  = y1.subM(y2);
132 
133     z  = S.z.mul(c);
134 
135     return new sjcl.ecc.pointJac(this.curve,x,y,z);
136   },
137   
138   /**
139    * doubles this point.
140    * @return {sjcl.ecc.pointJac} The doubled point.
141    */
142   doubl: function() {
143     if (this.isIdentity) { return this; }
144 
145     var
146       y2 = this.y.square(),
147       a  = y2.mul(this.x.mul(4)),
148       b  = y2.square().mul(8),
149       z2 = this.z.square(),
150       c  = this.curve.a.toString() == (new sjcl.bn(-3)).toString() ?
151                 this.x.sub(z2).mul(3).mul(this.x.add(z2)) :
152                 this.x.square().mul(3).add(z2.square().mul(this.curve.a)),
153       x  = c.square().subM(a).subM(a),
154       y  = a.sub(x).mul(c).subM(b),
155       z  = this.y.add(this.y).mul(this.z);
156     return new sjcl.ecc.pointJac(this.curve, x, y, z);
157   },
158 
159   /**
160    * Returns a copy of this point converted to affine coordinates.
161    * @return {sjcl.ecc.point} The converted point.
162    */
163   toAffine: function() {
164     if (this.isIdentity || this.z.equals(0)) {
165       return new sjcl.ecc.point(this.curve);
166     }
167     var zi = this.z.inverse(), zi2 = zi.square();
168     return new sjcl.ecc.point(this.curve, this.x.mul(zi2).fullReduce(), this.y.mul(zi2.mul(zi)).fullReduce());
169   },
170   
171   /**
172    * Multiply this point by k and return the answer in Jacobian coordinates.
173    * @param {bigInt} k The coefficient to multiply by.
174    * @param {sjcl.ecc.point} affine This point in affine coordinates.
175    * @return {sjcl.ecc.pointJac} The result of the multiplication, in Jacobian coordinates.
176    */
177   mult: function(k, affine) {
178     if (typeof(k) === "number") {
179       k = [k];
180     } else if (k.limbs !== undefined) {
181       k = k.normalize().limbs;
182     }
183     
184     var i, j, out = new sjcl.ecc.point(this.curve).toJac(), multiples = affine.multiples();
185 
186     for (i=k.length-1; i>=0; i--) {
187       for (j=sjcl.bn.prototype.radix-4; j>=0; j-=4) {
188         out = out.doubl().doubl().doubl().doubl().add(multiples[k[i]>>j & 0xF]);
189       }
190     }
191     
192     return out;
193   },
194   
195   /**
196    * Multiply this point by k, added to affine2*k2, and return the answer in Jacobian coordinates.
197    * @param {bigInt} k The coefficient to multiply this by.
198    * @param {sjcl.ecc.point} affine This point in affine coordinates.
199    * @param {bigInt} k2 The coefficient to multiply affine2 this by.
200    * @param {sjcl.ecc.point} affine The other point in affine coordinates.
201    * @return {sjcl.ecc.pointJac} The result of the multiplication and addition, in Jacobian coordinates.
202    */
203   mult2: function(k1, affine, k2, affine2) {
204     if (typeof(k1) === "number") {
205       k1 = [k1];
206     } else if (k1.limbs !== undefined) {
207       k1 = k1.normalize().limbs;
208     }
209     
210     if (typeof(k2) === "number") {
211       k2 = [k2];
212     } else if (k2.limbs !== undefined) {
213       k2 = k2.normalize().limbs;
214     }
215     
216     var i, j, out = new sjcl.ecc.point(this.curve).toJac(), m1 = affine.multiples(),
217         m2 = affine2.multiples(), l1, l2;
218 
219     for (i=Math.max(k1.length,k2.length)-1; i>=0; i--) {
220       l1 = k1[i] | 0;
221       l2 = k2[i] | 0;
222       for (j=sjcl.bn.prototype.radix-4; j>=0; j-=4) {
223         out = out.doubl().doubl().doubl().doubl().add(m1[l1>>j & 0xF]).add(m2[l2>>j & 0xF]);
224       }
225     }
226     
227     return out;
228   },
229 
230   isValid: function() {
231     var z2 = this.z.square(), z4 = z2.square(), z6 = z4.mul(z2);
232     return this.y.square().equals(
233              this.curve.b.mul(z6).add(this.x.mul(
234                this.curve.a.mul(z4).add(this.x.square()))));
235   }
236 };

238 /**
239  * Construct an elliptic curve. Most users will not use this and instead start with one of the NIST curves defined below.
240  *
241  * @constructor
242  * @param {bigInt} p The prime modulus.
243  * @param {bigInt} r The prime order of the curve.
244  * @param {bigInt} a The constant a in the equation of the curve y^2 = x^3 + ax + b (for NIST curves, a is always -3).
245  * @param {bigInt} x The x coordinate of a base point of the curve.
246  * @param {bigInt} y The y coordinate of a base point of the curve.
247  */
248 sjcl.ecc.curve = function(Field, r, a, b, x, y) {
249   this.field = Field;
250   this.r = new sjcl.bn(r);
251   this.a = new Field(a);
252   this.b = new Field(b);
253   this.G = new sjcl.ecc.point(this, new Field(x), new Field(y));
254 };
255 
256 sjcl.ecc.curve.prototype.fromBits = function (bits) {
257   var w = sjcl.bitArray, l = this.field.prototype.exponent + 7 & -8,
258       p = new sjcl.ecc.point(this, this.field.fromBits(w.bitSlice(bits, 0, l)),
259                              this.field.fromBits(w.bitSlice(bits, l, 2*l)));
260   if (!p.isValid()) {
261     throw new sjcl.exception.corrupt("not on the curve!");
262   }
263   return p;
264 };

325 /** our basicKey classes
326 */
327 sjcl.ecc.basicKey = {
328   /** ecc publicKey. 
329   * @constructor
330   * @param {curve} curve the elliptic curve
331   * @param {point} point the point on the curve
332   */
333   publicKey: function(curve, point) {
334     this._curve = curve;
335     this._curveBitLength = curve.r.bitLength();
336     if (point instanceof Array) {
337       this._point = curve.fromBits(point);
338     } else {
339       this._point = point;
340     }
341 
342     /** get this keys point data
343     * @return x and y as bitArrays
344     */
345     this.get = function() {
346       var pointbits = this._point.toBits();
347       var len = sjcl.bitArray.bitLength(pointbits);
348       var x = sjcl.bitArray.bitSlice(pointbits, 0, len/2);
349       var y = sjcl.bitArray.bitSlice(pointbits, len/2);
350       return { x: x, y: y };
351     };
352   },
353 
354   /** ecc secretKey
355   * @constructor
356   * @param {curve} curve the elliptic curve
357   * @param exponent
358   */
359   secretKey: function(curve, exponent) {
360     this._curve = curve;
361     this._curveBitLength = curve.r.bitLength();
362     this._exponent = exponent;
363 
364     /** get this keys exponent data
365     * @return {bitArray} exponent
366     */
367     this.get = function () {
368       return this._exponent.toBits();
369     };
370   }
371 };

373 /** @private */
374 sjcl.ecc.basicKey.generateKeys = function(cn) {
375   return function generateKeys(curve, paranoia, sec) {
376     curve = curve || 256;
377     paranoia = paranoia || 0;
378 
379     if (typeof curve === "number") {
380       curve = sjcl.ecc.curves['c'+curve];
381       if (curve === undefined) {
382         throw new sjcl.exception.invalid("no such curve");
383       }
384     }
385     sec = sec || sjcl.bn.random(curve.r, paranoia);
386 
387     var pub = curve.G.mult(sec);
388     return { pub: new sjcl.ecc[cn].publicKey(curve, pub),
389              sec: new sjcl.ecc[cn].secretKey(curve, sec) };
390   };
391 };

418 sjcl.ecc.elGamal.publicKey.prototype = {
419   /** Kem function of elGamal Public Key
420   * @param paranoia paranoia to use for randomization.
421   * @return {object} key and tag. unkem(tag) with the corresponding secret key results in the key returned.
422   */
423   kem: function(paranoia) {
424     var sec = sjcl.bn.random(this._curve.r, paranoia),
425         tag = this._curve.G.mult(sec).toBits(),
426         key = sjcl.hash.sha256.hash(this._point.mult(sec).toBits());
427     return { key: key, tag: tag };
428   }
429 };

432   /** UnKem function of elGamal Secret Key
433   * @param {bitArray} tag The Tag to decrypt.
434   * @return {bitArray} decrypted key.
435   */
436   unkem: function(tag) {
437     return sjcl.hash.sha256.hash(this._curve.fromBits(tag).mult(this._exponent).toBits());
438   },

440   /** Diffie-Hellmann function
441   * @param {elGamal.publicKey} pk The Public Key to do Diffie-Hellmann with

443   */
444   dh: function(pk) {
445     return sjcl.hash.sha256.hash(pk._point.mult(this._exponent).toBits());

468 /** specific functions for ecdsa publicKey. */
469 sjcl.ecc.ecdsa.publicKey.prototype = {
470   /** Diffie-Hellmann function
471   * @param {bitArray} hash hash to verify. 
472   * @param {bitArray} rs signature bitArray.
473   * @param {boolean}  fakeLegacyVersion use old legacy version
474   */
475   verify: function(hash, rs, fakeLegacyVersion) {
476     if (sjcl.bitArray.bitLength(hash) > this._curveBitLength) {
477       hash = sjcl.bitArray.clamp(hash, this._curveBitLength);
478     }
479     var w = sjcl.bitArray,
480         R = this._curve.r,
481         l = this._curveBitLength,
482         r = sjcl.bn.fromBits(w.bitSlice(rs,0,l)),
483         ss = sjcl.bn.fromBits(w.bitSlice(rs,l,2*l)),
484         s = fakeLegacyVersion ? ss : ss.inverseMod(R),
485         hG = sjcl.bn.fromBits(hash).mul(s).mod(R),
486         hA = r.mul(s).mod(R),
487         r2 = this._curve.G.mult2(hG, hA, this._point).x;
488     if (r.equals(0) || ss.equals(0) || r.greaterEquals(R) || ss.greaterEquals(R) || !r2.equals(r)) {
489       if (fakeLegacyVersion === undefined) {
490         return this.verify(hash, rs, true);
491       } else {
492         throw (new sjcl.exception.corrupt("signature didn't check out"));
493       }
494     }
495     return true;
496   }
497 };

507 /** specific functions for ecdsa secretKey. */
508 sjcl.ecc.ecdsa.secretKey.prototype = {
509   /** Diffie-Hellmann function
510   * @param {bitArray} hash hash to sign. 
511   * @param {int} paranoia paranoia for random number generation
512   * @param {boolean} fakeLegacyVersion use old legacy version
513   */
514   sign: function(hash, paranoia, fakeLegacyVersion, fixedKForTesting) {
515     if (sjcl.bitArray.bitLength(hash) > this._curveBitLength) {
516       hash = sjcl.bitArray.clamp(hash, this._curveBitLength);
517     }
518     var R  = this._curve.r,
519         l  = R.bitLength(),
520         k  = fixedKForTesting || sjcl.bn.random(R.sub(1), paranoia).add(1),
521         r  = this._curve.G.mult(k).x.mod(R),
522         ss = sjcl.bn.fromBits(hash).add(r.mul(this._exponent)),
523         s  = fakeLegacyVersion ? ss.inverseMod(R).mul(k).mod(R)
524              : ss.mul(k.inverseMod(R)).mod(R);
525     return sjcl.bitArray.concat(r.toBits(l), s.toBits(l));
526   }
